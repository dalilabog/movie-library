import tkinter as tk
from tkinter import ttk, messagebox
import json
import os

class MovieLibraryApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Movie Library")
        self.root.geometry("800x600")
        
        self.db_file = "movies.json"
        self.movies = self.load_data()

        # Поля ввода
        input_frame = tk.LabelFrame(root, text="Добавить новый фильм", padx=10, pady=10)
        input_frame.pack(fill="x", padx=10, pady=5)

        tk.Label(input_frame, text="Название:").grid(row=0, column=0)
        self.title_entry = tk.Entry(input_frame)
        self.title_entry.grid(row=0, column=1, padx=5)

        tk.Label(input_frame, text="Жанр:").grid(row=0, column=2)
        self.genre_combobox = ttk.Combobox(input_frame, values=["Боевик", "Комедия", "Драма", "Ужасы", "Фантастика"], state="readonly")
        self.genre_combobox.grid(row=0, column=3, padx=5)

        tk.Label(input_frame, text="Год:").grid(row=1, column=0, pady=5)
        self.year_entry = tk.Entry(input_frame)
        self.year_entry.grid(row=1, column=1, padx=5)

        tk.Label(input_frame, text="Рейтинг (0-10):").grid(row=1, column=2)
        self.rating_entry = tk.Entry(input_frame)
        self.rating_entry.grid(row=1, column=3, padx=5)

        add_btn = tk.Button(input_frame, text="Добавить", command=self.add_movie, bg="#4CAF50", fg="white", width=15)
        add_btn.grid(row=0, column=4, rowspan=2, padx=20)

        # Фильтры
        filter_frame = tk.Frame(root, padx=10, pady=5)
        filter_frame.pack(fill="x")
        tk.Label(filter_frame, text="Фильтр Жанр:").pack(side="left")
        self.filter_genre = ttk.Combobox(filter_frame, values=["Все", "Боевик", "Комедия", "Драма", "Ужасы", "Фантастика"], state="readonly")
        self.filter_genre.current(0)
        self.filter_genre.pack(side="left", padx=5)
        self.filter_genre.bind("<<ComboboxSelected>>", lambda e: self.update_table())

        # Таблица
        self.table = ttk.Treeview(root, columns=("title", "genre", "year", "rating"), show="headings")
        for col in ("title", "genre", "year", "rating"):
            self.table.heading(col, text=col.capitalize())
        self.table.pack(fill="both", expand=True, padx=10, pady=10)

        self.update_table()

    def load_data(self):
        if not os.path.exists(self.db_file):
            return []
        try:
            with open(self.db_file, "r", encoding="utf-8") as f:
                return json.load(f)
        except (json.JSONDecodeError, OSError) as e:
            messagebox.showerror("Ошибка загрузки", f"Не удалось прочитать базу данных: {e}")
            return []

    def add_movie(self):
        title = self.title_entry.get().strip()
        genre = self.genre_combobox.get()
        year_raw = self.year_entry.get().strip()
        rating_raw = self.rating_entry.get().strip().replace(',', '.')

        # Строгая валидация (Шаг 5+)
        if not all([title, genre, year_raw, rating_raw]):
            messagebox.showwarning("Внимание", "Все поля должны быть заполнены!")
            return

        if not year_raw.isdigit():
            messagebox.showerror("Ошибка", "Год должен содержать только цифры!")
            return

        try:
            rating = float(rating_raw)
            if not (0 <= rating <= 10):
                raise ValueError
        except ValueError:
            messagebox.showerror("Ошибка", "Рейтинг должен быть числом от 0.0 до 10.0!")
            return

        self.movies.append({"title": title, "genre": genre, "year": int(year_raw), "rating": rating})
        self.save_data()
        self.update_table()
        
        # Очистка
        self.title_entry.delete(0, tk.END)
        self.year_entry.delete(0, tk.END)
        self.rating_entry.delete(0, tk.END)

    def update_table(self):
        for item in self.table.get_children():
            self.table.delete(item)
        f_genre = self.filter_genre.get()
        for m in self.movies:
            if f_genre == "Все" or m['genre'] == f_genre:
                self.table.insert("", "end", values=(m['title'], m['genre'], m['year'], m['rating']))

    def save_data(self):
        try:
            with open(self.db_file, "w", encoding="utf-8") as f:
                json.dump(self.movies, f, indent=4, ensure_ascii=False)
        except OSError as e:
            messagebox.showerror("Ошибка сохранения", f"Не удалось записать файл: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MovieLibraryApp(root)
    root.mainloop()
